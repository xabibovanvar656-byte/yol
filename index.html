-- ============================================ -- Yoʻl — Комнаты и дуэль по ID друга -- Запустить после duels_setup.sql -- ============================================ -- поле кода комнаты и тип дуэли alter table duels add column if not exists room_code text default ''; alter table duels add column if not exists target_email text default ''; alter table duels add column if not exists duel_type text default 'random'; create index if not exists idx_duels_room on duels(room_code); create index if not exists idx_duels_target on duels(target_email); -- RPC: создать комнату с кодом create or replace function create_room(p_email text, p_name text, p_questions text, p_code text) returns bigint language plpgsql security definer as $$ declare new_id bigint; begin insert into duels(player1_email,player1_name,status,questions_json,room_code,duel_type) values(p_email,p_name,'waiting',p_questions,p_code,'room') returning id into new_id; return new_id; end; $$; -- RPC: войти в комнату по коду create or replace function join_room(p_email text, p_name text, p_code text) returns bigint language plpgsql security definer as $$ declare room_id bigint; begin select id into room_id from duels where room_code=p_code and status='waiting' and player1_email <> p_email order by created_at desc limit 1; if room_id is null then return 0; end if; update duels set player2_email=p_email, player2_name=p_name, status='active' where id=room_id; return room_id; end; $$; -- RPC: вызвать друга по email create or replace function challenge_friend(p_email text, p_name text, p_questions text, p_target text) returns bigint language plpgsql security definer as $$ declare new_id bigint; begin insert into duels(player1_email,player1_name,status,questions_json,target_email,duel_type) values(p_email,p_name,'waiting',p_questions,p_target,'friend') returning id into new_id; return new_id; end; $$; grant execute on function create_room(text,text,text,text) to anon; grant execute on function join_room(text,text,text) to anon; grant execute on function challenge_friend(text,text,text,text) to anon; select 'Комнаты и дуэль с другом настроены!' as result;
